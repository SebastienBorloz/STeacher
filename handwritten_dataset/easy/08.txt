\[

\]

\[
P(\{1,1\}) = \frac{4}{16} \binom{4}{1} = \frac{1}{6}
\]

\[
P(\{2,2\}) = \frac{6}{16} \binom{4}{2} = a
\]

\[
P(\{3,3\}) = \frac{\binom{4}{3}}{16} = \frac{(4-3)! \cdot 3!}{16} = \frac{4}{16}
\]

\[
P(\{4,0\}) = \frac{\binom{4}{4}}{16} = \frac{1}{16}
